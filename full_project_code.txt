const express = require('express');
const router = express.Router();

const { readTrafficData } = require('../utils/readTrafficData');

/**
 * GET /api/traffic/all
 */
router.get('/all', (req, res) => {
    try {
        const data = readTrafficData();
        res.json(data);
    } catch (err) {
        res.status(500).json({ error: 'Failed to read traffic data' });
    }
});

/**
 * GET /api/traffic/latest
 */
router.get('/latest', (req, res) => {
    try {
        const data = readTrafficData();
        const latestByRoute = {};

        data.forEach(entry => {
            const key = `${entry.origin}__${entry.destination}`;
            latestByRoute[key] = entry;
        });

        res.json(Object.values(latestByRoute));
    } catch (err) {
        res.status(500).json({ error: 'Failed to compute latest traffic' });
    }
});

/**
 * GET /api/traffic/top-congested
 */
router.get('/top-congested', (req, res) => {
    try {
        const data = readTrafficData();

        // Step 1: latest record per route
        const latestByRoute = {};
        data.forEach(entry => {
            const key = `${entry.origin}__${entry.destination}`;
            latestByRoute[key] = entry;
        });

        // Step 2: sort by delay percentage
        const sorted = Object.values(latestByRoute).sort(
            (a, b) => b.delay_percent - a.delay_percent
        );

        // Step 3: return top 5
        res.json(sorted.slice(0, 5));
    } catch (err) {
        res.status(500).json({ error: 'Failed to compute top congested routes' });
    }
});

module.exports = router;

const DATA_PATH = path.join(__dirname, '../data/traffic-log.json');

function readTrafficData() {
    const raw = fs.readFileSync(DATA_PATH, 'utf-8');

}

import fetch from 'node-fetch';
import fs from 'fs';
import dotenv from 'dotenv';
import locations from './locations.js';

dotenv.config();

const API_KEY = process.env.GOOGLE_MAPS_API_KEY;
const URL = 'https://routes.googleapis.com/directions/v2:computeRoutes';

function parseDuration(str) {
    if (str.endsWith('s')) return parseInt(str);
    const m = str.match(/PT(?:(\d+)M)?(?:(\d+)S)?/);
    return (parseInt(m[1] || 0) * 60) + parseInt(m[2] || 0);
}

async function run() {
    for (const route of locations) {
        const body = {
            origin: route.origin,
            destination: route.destination,
            travelMode: "DRIVE",
            routingPreference: "TRAFFIC_AWARE_OPTIMAL",
            departureTime: new Date(Date.now() + 300000).toISOString()
        };

        const res = await fetch(URL, {
            method: 'POST',
            headers: {
                'Content-Type': 'application/json',
                'X-Goog-Api-Key': API_KEY,
                'X-Goog-FieldMask': 'routes.duration,routes.distanceMeters'
            },
            body: JSON.stringify(body)
        });

        const data = await res.json();
        const rt = parseDuration(data.routes[0].duration);
        const delay = ((rt - route.freeflow_time * 60) / (route.freeflow_time * 60) * 100).toFixed(1);

        const log = {
            time: new Date().toISOString(),
            origin: route.origin.address,
            destination: route.destination.address,
            delay_percent: delay
        };

        fs.appendFileSync('traffic-log.json', JSON.stringify(log) + '\n');
        console.log(log);
    }
}

run();
